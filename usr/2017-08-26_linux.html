下面是bash配置，方便在终端中使用，还是有些乱，准备写一个bash脚本来统一管理 - vi ~/.bashrc ```shell # Fox @ 2017-08-21 # export PATH="$HOME/bin:$PATH" # export _JAVA_OPTIONS='-Dawt.useSystemAAFontSettings=on' alias lh='ls -lh' alias jj='java -jar' alias bb='java -jar FoxBook_Core.jar' alias mm='free -m' alias du='du -sh' alias n='novelManager.sh' alias sshwrt='ssh root@10.2.3.1' getwrt(){ wget http://10.2.3.1:8080/k4/$1 ; } putwrt(){ curl http://10.2.3.1:8080/cgi-bin/ff.lua -F f=@$1 ; } flv2mp4() { ffmpeg -i $1.flv -vcodec copy -acodec copy -movflags faststart $1.mp4 ; } [ -d /dev/shm/00 ] || mkdir /dev/shm/00 ``` ## GUI环境的一些设置 -
